Minimal example

   ggplot(iris, aes(x = Sepal.Width)) +
   geom_bar()  


   plot <- ggplot(iris, aes(x = Sepal.Width)) 
   plot <- plot + geom_bar()  

   plot

Data

   ggplot(iris[1:10, ], aes(x = Sepal.Width)) +
   geom_bar()  


   ggplot(subset(iris, Species == "setosa"), aes(x = Sepal.Width)) +
   geom_bar()  


   require(dplyr)

   iris %>%
   filter(Species == "virginica") %>%

   ggplot(aes(x = Sepal.Width)) +
   geom_bar()
